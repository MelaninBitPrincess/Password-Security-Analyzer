5    class PasswordAnalyzer:
6        def __init__(self):
7            self.common_passwords = self.load_common_passwords()

9        def load_common_passwords(self):
10           """Load common passwords from file"""
11           try:
12               with open('common_passwords.txt', 'r') as f:
13                   return set(line.strip().lower() for line in f)
14           except FileNotFoundError:
15               return set(['password', '123456', 'admin', 'welcome'])
16   
17       def check_length(self, password):
18           """Check if password meets minimum length requirement"""
19           return len(password) >= 8
20   
21       def check_complexity(self, password):
22           """Check password complexity requirements"""
23           checks = {
24               'uppercase': bool(re.search(r'[A-Z]', password)),
25               'lowercase': bool(re.search(r'[a-z]', password)),
26               'digits': bool(re.search(r'\d', password)),
27               'special_chars': bool(re.search(r'[!@#$%^&*(),.?":{}|<>]', password))
28           }
29           return checks
30   
31       def check_common_patterns(self, password):
32           """Check for common weak patterns"""
33           patterns = {
34               'sequential_numbers': bool(re.search(r'123|234|345|456|567|678|789', password)),
35               'keyboard_patterns': bool(re.search(r'qwerty|asdf|zxcv', password.lower())),
36               'repeated_chars': bool(re.search(r'(.)\1{2,}', password))
37           }
38           return patterns
39   
40       def check_against_common_list(self, password):
41           """Check if password is in common passwords list"""
42           return password.lower() in self.common_passwords
43   
44       def calculate_entropy(self, password):
45           """Calculate password entropy (simplified)"""
46           charset_size = 0
47           if re.search(r'[a-z]', password): charset_size += 26
48           if re.search(r'[A-Z]', password): charset_size += 26
49           if re.search(r'\d', password): charset_size += 10
50           if re.search(r'[!@#$%^&*(),.?":{}|<>]', password): charset_size += 32
51   
52           if charset_size == 0: return 0
53           import math
54           return len(password) * math.log2(charset_size)
55   
56       def hash_password(self, password):
57           """Demonstrate password hashing"""
58           return hashlib.sha256(password.encode()).hexdigest()
59   
60       def analyze_password(self, password):
61           """Comprehensive password analysis"""
62           results = {
63               'password': password,
64               'length_ok': self.check_length(password),
65               'complexity': self.check_complexity(password),
66               'weak_patterns': self.check_common_patterns(password),
67               'is_common': self.check_against_common_list(password),
68               'entropy': self.calculate_entropy(password),
69               'hash': self.hash_password(password)
70           }
71   
72           # Calculate overall score
73           score = 0
74           if results['length_ok']: score += 20
75           score += sum(results['complexity'].values()) * 15
76           if not any(results['weak_patterns'].values()): score += 20
77           if not results['is_common']: score += 20
78   
79           results['score'] = min(score, 100)
80           results['strength'] = self.get_strength_rating(results['score'])
81   
82           return results
83   
84       def get_strength_rating(self, score):
85           """Convert score to strength rating"""
86           if score < 30:
87               return "Very Weak"
88           elif score < 50:
89               return "Weak"
90           elif score < 70:
91               return "Fair"
92           elif score < 90:
93               return "Good"
94           else:
95               return "Strong"
